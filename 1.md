//   {"modelName":"resnet3.0", "timeWindow":"2025041010"}

for (String modelName : obj.keySet()) {                // ← 모든 키 다 돌리기
    JsonArray val = obj.getAsJsonArray(modelName);
}
🧪 직접 따라 써보기 좋은 미니 예제
java// 가짜 JSON 만들어서 테스트
String fakeJson = "{\"a\":[\"x\",\"y\"], \"b\":[\"z\"]}";
JsonObject obj = new Gson().fromJson(fakeJson, JsonObject.class);

System.out.println("키 개수: " + obj.keySet().size());     // 2

for (String key : obj.keySet()) {
    JsonArray arr = obj.getAsJsonArray(key);
    System.out.print(key + " = [");
    for (int i = 0; i < arr.size(); i++) {
        if (i > 0) System.out.print(",");
        System.out.print(arr.get(i).getAsString());
    }
    System.out.println("]");
}
출력:
키 개수: 2
a = [x,y]
b = [z]
